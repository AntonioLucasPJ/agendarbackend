create table users (
    id_user INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(50),
    email VARCHAR(100),
    endereço VARHCAR(100),
    telefone VARCHAR(100),
    password VARCHAR(100)
);

create table admins (
    id_admin INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR (100),
    email VARCHAR(100),
    password VARCHAR(100)
);
CREATE TABLE IF NOT EXISTS mecanicos (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    nome VARCHAR(100) NOT NULL,
    cpf VARCHAR(14) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    telefone VARCHAR(20),
    titulo_profissional VARCHAR(50) NOT NULL DEFAULT 'Mecânico Geral',
    tempo_experiencia INTEGER NOT NULL DEFAULT 0,
    foto_url TEXT,
    descricao TEXT,
    status CHAR(1) NOT NULL DEFAULT 'A', -- 'A' para Ativo, 'I' para Inativo
    criado_em TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE IF NOT EXISTS services (
    id_service INTEGER PRIMARY KEY AUTOINCREMENT,
    description VARCHAR (50)
);
CREATE TABLE IF NOT EXISTS mecanicos_services (
    id_mecanico_service INTEGER PRIMARY KEY AUTOINCREMENT,
    id_mecanico INTEGER,
    id_service INTEGER,
    foreign key (id_mecanico) references mecanicos(id),
    foreign key (id_service) references services(id_service)
);
CREATE TABLE IF NOT EXISTS appointments (
    id INTEGER PRIMARY KEY AUTOINCREMENT, -- No SQLite usa-se INTEGER para autoincremento
    id_user INTEGER NOT NULL,
    id_service INTEGER NOT NULL,
    vehicle_model TEXT NOT NULL,
    license_plate TEXT NOT NULL,
    booking_date TEXT NOT NULL,       -- SQLite não tem tipo DATE, salvamos como TEXT (YYYY-MM-DD)
    booking_hour TEXT NOT NULL,
    horario_inicio TEXT NOT NULL,         -- Salvamos como TEXT (HH:MM:SS)
    horario_fim TEXT NOT NULL,
    observacoes TEXT,
    status TEXT DEFAULT 'P',
    created_at TEXT DEFAULT CURRENT_TIMESTAMP, -- Funciona automaticamente no INSERT
    updated_at TEXT DEFAULT CURRENT_TIMESTAMP,
    
    FOREIGN KEY (id_user) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (id_service) REFERENCES services(id)
);

CREATE TABLE IF NOT EXISTS appointment_services(
    id_appointment_service INTEGER PRIMARY KEY AUTOINCREMENT,
    id_appointment INTEGER,
    id_service INTEGER,
    price_charged DECIMAL(9,2),
    FOREIGN KEY (id_appointment) REFERENCES appointments(id_appointment) ON DELETE CASCADE,
    FOREIGN KEY (id_service) REFERENCES services(id_service)
);    

-- 1. Tabela de Marcas (Brands)
CREATE TABLE IF NOT EXISTS brands (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE,
    image_url TEXT
);

-- 2. Tabela de Modelos (Vehicle_Models) - Apontando para 'brands'
CREATE TABLE vehicle_models (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    brands_id INTEGER NOT NULL,
    name TEXT NOT NULL,
    image_url NOT NULL,
    year INTEFER NOT NULL,
    FOREIGN KEY (brands_id) REFERENCES brands(id) ON DELETE CASCADE
);


-- 3. Tabela de Veículos do Cliente - Corrigida para apontar para 'vehicle_models'
CREATE TABLE vehicle_client (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    usuario_id INTEGER NOT NULL,
    modelo_id INTEGER NOT NULL,
    placa TEXT NOT NULL UNIQUE,
    cor TEXT NOT NULL,
    -- O ERRO ESTAVA AQUI: Deve apontar para vehicle_models(id), e não para models(id)
    FOREIGN KEY (modelo_id) REFERENCES vehicle_models(id), 
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
