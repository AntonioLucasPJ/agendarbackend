create table users (
    id_user INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(50),
    email VARCHAR(100),
    telefone VARCHAR(100),
    password VARCHAR(100)
);

create table admins (
    id_admin INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR (100),
    email VARCHAR(100),
    password VARCHAR(100)
);
create table mecanicos (
    id_mecanico INTEGER PRIMARY KEY AUTOINCREMENT,
    name VARCHAR(50),
    specialty VARCHAR(50),
    icon VARCHAR(50)
);
create table services (
    id_service INTEGER PRIMARY KEY AUTOINCREMENT,
    description VARCHAR (50)
);
create table mecanicos_services (
    id_mecanico_service INTEGER PRIMARY KEY AUTOINCREMENT,
    id_mecanico INTEGER,
    id_service INTEGER,
    price     DECIMAL(9,2),
    foreign key (id_mecanico) references mecanicos(id_mecanico),
    foreign key (id_service) references services(id_service)
);
create table appointments(
    id_appointment INTEGER PRIMARY KEY AUTOINCREMENT,
    id_mecanico INTEGER,
    id_service INTERGER,
    id_user INTEGER,
    booking_date DATE,
    booking_hour VARCHAR(5),
    foreign key (id_mecanico) references mecanicos(id_mecanico),
    foreign key (id_service) references services(id_service),
    foreign key (id_user) references users(id_user)
    
);


-- 1. Tabela de Marcas (Brands)
CREATE TABLE brands (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    name TEXT NOT NULL UNIQUE
);

-- 2. Tabela de Modelos (Vehicle_Models) - Apontando para 'brands'
CREATE TABLE vehicle_models (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    brands_id INTEGER NOT NULL,
    name TEXT NOT NULL,
    FOREIGN KEY (brands_id) REFERENCES brands(id) ON DELETE CASCADE
);

-- 3. Tabela de Veículos do Cliente - Corrigida para apontar para 'vehicle_models'
CREATE TABLE veiculos_cliente (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    usuario_id INTEGER NOT NULL,
    modelo_id INTEGER NOT NULL,
    placa TEXT NOT NULL UNIQUE,
    cor TEXT NOT NULL,
    -- O ERRO ESTAVA AQUI: Deve apontar para vehicle_models(id), e não para models(id)
    FOREIGN KEY (modelo_id) REFERENCES vehicle_models(id), 
    FOREIGN KEY (usuario_id) REFERENCES usuarios(id) ON DELETE CASCADE
);
